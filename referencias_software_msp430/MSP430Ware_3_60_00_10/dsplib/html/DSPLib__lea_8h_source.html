1154 static inline bool msp_lea_acquireLock(void)
1155 {
1156 #ifndef MSP_DISABLE_DIAGNOSTICS
1157  bool status;
1158 
1159  /* Save interrupt state and disable interrupts. */
1160  uint16_t interruptState = __get_interrupt_state();
1161  __disable_interrupt();
1162 
1163  /* Check LEA-SC busy flag and DSPLib lock flags. */
1164  if ((LEACNF1 & LEABUSY) || msp_lea_locked) {
1165  status = false;
1166  }
1167  else {
1168  status = true;
1169  msp_lea_locked = true;
1170  }
1171 
1172  /* Restore interrupt state and return status. */
1173  __set_interrupt_state(interruptState);
1174  return status;
1175 #endif //MSP_DISABLE_DIAGNOSTICS
1176 }
